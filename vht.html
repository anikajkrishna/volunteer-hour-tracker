    <script>

        // Dynamically create the grid (12x31)

        const chart = document.getElementById('chart');

        for (let i = 0; i < 12 * 31; i++) {

            const box = document.createElement('div');

            chart.appendChild(box);

        }


        const chartBoxes = document.querySelectorAll('.chart div');

        const popup = document.getElementById('popup');

        const overlay = document.getElementById('overlay');

        const dateInput = document.getElementById('volunteer-date');

        let currentBox = null;


        chartBoxes.forEach((box, index) => {

            box.addEventListener('click', () => {

                currentBox = box;

                const day = (index % 31) + 1;

                const month = Math.floor(index / 31) + 1;

                const date = `${month}/${day}`;

                dateInput.value = date;  // Set the date in the popup

                popup.style.display = 'block';  // Show popup

                overlay.style.display = 'block';  // Show overlay

            });

        });


        function saveDetails() {

            const location = document.getElementById('volunteer-location').value;

            let color;


            // Assign colors based on the selected location

            switch(location) {

                case "FMSC 1 session":

                    color = "#E36650";

                    break;

                case "FMSC 2 sessions":

                    color = "#C19EDC";

                    break;

                case "FMSC 3 sessions":

                    color = "#B0D492";

                    break;

                case "FMSC 4 sessions":

                    color = "#7EB6C2";

                    break;

                case "Metrocrest Orientation":

                    color = "#B1B1B1";

                    break;

                case "Metrocrest Pantry 1 session":

                    color = "#6EA3D7";

                    break;

                case "Metrocrest Pantry 2 sessions":

                    color = "#8E6953";

                    break;

                case "Metrocrest Resale 1 session":

                    color = "#F9BE84";

                    break;

                case "Better Impact 1 hour":

                case "Better Impact 2 hours":

                case "Notelove Lesson":

                    break;

                default:

                    color = "#f0f0f0"; // Default color

            }


            // Set the background color of the clicked box

            if (currentBox) {

                currentBox.style.backgroundColor = color;

            }

        }
